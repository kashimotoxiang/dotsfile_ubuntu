apt-get update -y && apt-get install curl install vim unzip wget curl git zsh tmux htop fzf fd-find tldr -y

yes | sh -c "$(curl -fsSL https://raw.githubusercontent.com/kashimotoxiang/dotsfile_ubuntu/master/install.sh)"
